import ray
import pymysql
from ray.data import read_parquet

ray.init()

# 读 S3 历史
ds_history = read_parquet("s3://my-lake/raw/*.parquet")

# 读 Doris 新鲜
def get_doris_conn():
    return pymysql.connect(host="doris-fe", port=9030, user="root", db="dw")
ds_fresh = ray.data.read_sql("SELECT * FROM fresh_data", get_doris_conn)

# 合并
ds = ds_history.union(ds_fresh)

# 2. Great Expectations 质量门禁
import great_expectations as gx
from great_expectations.dataset import PandasDataset

def validate_batch(batch):
    ds_ge = PandasDataset(batch)
    ds_ge.expect_column_values_to_not_be_null("user_id")
    ds_ge.expect_column_values_to_be_between("age", 0, 120)
    result = ds_ge.validate()
    if not result["success"]:
        raise ValueError(f"质量失败: {result}")
    return batch

ds_validated = ds.map_batches(validate_batch, batch_format="pandas")

# 3. whylogs 轻量日志
import whylogs as why
from whylogs.api.logger import log

profile = log(ds_validated.to_pandas()).profile()
profile.write("s3://my-lake/whylogs/raw_profile.bin")

import mlflow
from ray.train.torch import TorchTrainer
from ray.air.integrations.mlflow import MLflowLoggerCallback
from ray.tune import Tuner, grid_search

mlflow.set_tracking_uri("http://mlflow:5000")

def train_loop(config):
    import torch
    from torch.utils.data import DataLoader
    from ray.train import Checkpoint, get_checkpoint
    
    # 训练逻辑
    model = torch.nn.Linear(10, 1)
    optimizer = torch.optim.Adam(model.parameters(), lr=config["lr"])
    
    # 加载 Feast 特征
    from feast import FeatureStore
    store = FeatureStore(repo_path=".")
    training_df = store.get_historical_features(
        entity_df="SELECT user_id, label, event_ts FROM entities",
        features=["user_features:age", "user_features:income"]
    ).to_df()
    
    # 训练
    for epoch in range(10):
        loss = 0.1  # 模拟
        mlflow.log_metric("loss", loss, step=epoch)
    
    # 保存模型
    mlflow.pytorch.log_model(model, "model")

# 分布式训练器
trainer = TorchTrainer(
    train_loop,
    scaling_config={"num_workers": 4, "use_gpu": True},
    datasets={"train": ds_processed},
    preprocessor=preprocessor
)

# 超参优化
tuner = Tuner(
    trainer,
    param_space={"lr": grid_search([1e-3, 1e-4])},
    run_config={"callbacks": [MLflowLoggerCallback(save_artifact=True)]}
)

results = tuner.fit()

from ray import serve
from fastapi import FastAPI
import mlflow

app = FastAPI()

@serve.deployment(num_replicas=2, ray_actor_options={"num_gpus": 0.5})
@serve.ingress(app)
class ModelDeployment:
    def __init__(self, model_uri):
        self.model = mlflow.pytorch.load_model(model_uri)
        self.preprocessor = preprocessor  # 复用预处理
    
    @app.post("/predict")
    def predict(self, data: dict):
        # 从 Feast 取在线特征
        from feast import FeatureStore
        store = FeatureStore(repo_path=".")
        features = store.get_online_features(
            features=["user_features:age", "user_features:income"],
            entity_rows=[{"user_id": data["user_id"]}]
        ).to_dict()
        
        # 预处理 + 预测
        processed = self.preprocessor.transform_batch(features)
        return {"prediction": float(self.model(processed))}

# 部署 Staging 版本
staging_uri = "models:/UserModel/Staging"
deployment = ModelDeployment.bind(staging_uri)
serve.run(deployment, name="user-model")

prod_uri = "models:/UserModel/Production"
new_uri = "models:/UserModel/2"

@serve.deployment
class TrafficSplit:
    def __init__(self, old_deployment, new_deployment, split=0.1):
        self.old = old_deployment
        self.new = new_deployment
        self.split = split
    
    async def __call__(self, request):
        import random
        if random.random() < self.split:
            return await self.new(request)
        else:
            return await self.old(request)

old_deployment = ModelDeployment.bind(prod_uri)
new_deployment = ModelDeployment.bind(new_uri)
traffic_split = TrafficSplit.bind(old_deployment, new_deployment, split=0.1)
serve.run(traffic_split, name="user-model-split")
